```sql
CREATE OR REPLACE PACKAGE BODY pkg_demo IS
  -- Private variable
  v_counter NUMBER := 0;

  PROCEDURE add_order(p_id NUMBER, p_name VARCHAR2) IS
  BEGIN
    INSERT INTO orders(order_id, order_name)
    VALUES (p_id, p_name);
    v_counter := v_counter + 1;
  END;

  PROCEDURE list_orders(p_cur OUT t_cursor) IS
  BEGIN
    OPEN p_cur FOR
      SELECT * FROM orders;
  END;

END pkg_demo;
/
```
